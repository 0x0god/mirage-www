let echo () =
  lwt mgr, mgr_t = Manager.create () in
  let src = None, 8081 in 
  Flow.listen mgr (`TCPv4 (src,
    (fun (addr, port) t ->
       Console.log "From %s:%d" (ipv4_addr_to_string addr) port);
       let rec echo () =
         lwt res = Flow.read t in
         match res with
         |None ->
            Console.log "Connection closed";
            return ()
         |Some data ->
            Flow.write t data >>= echo
       in
       echo ()
    )
  ))
